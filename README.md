Первый этап завершён. Осталось понять как результат умножение перевести в нужную  мне систему.

Обработка нуля: если число равно 0, результат — "0".
Обработка отрицательных чисел: запомнить знак, работать с модулем числа.
Цикл деления:
Делим число на 18.
Остаток от деления определяет текущую цифру:
0–9 → символы '0'–'9'
10–17 → символы 'A'–'H'
Добавляем цифру в строку (в обратном порядке).
Повторяем, пока число не станет равным 0.
Разворот строки: так как цифры собираются от младших разрядов к старшим, строку нужно развернуть.
Добавление знака: если исходное число было отрицательным, добавляем '-' в начало


#include <iostream>
#include <string>
#include <algorithm>
#include <iomanip>

using namespace std;

// Функция для перевода числа из десятичной в 18-ричную систему счисления
string decimalToBase18(int decimal) {			//Объявление функции, которая принимает целое число в десятичной системе и возвращает его представление в 18-ричной системе как строку.

    if (decimal == 0) return "0";              // Обработка особого случая: если входное число равно нулю, сразу возвращаем строку "0".

    const string digits = "0123456789ABCDEFGH"; // Символы для 18-ричной системы 	// Строка-шаблон, где каждый символ соответствует цифре в 18-ричной системе:
    string result = ""; 						// Инициализация пустой строки для накопления результата (цифры будут добавляться в обратном порядке)

    while (decimal > 0) {						// Цикл продолжается, пока число не станет равным нулю (алгоритм последовательного деления).
        int remainder = decimal % 18;
        result += digits[remainder];
        decimal /= 18;
    }

    reverse(result.begin(), result.end());
    return result;
}

int main() {
    for (int i = 1; i <= 10; i++) {
        for (int j = 1; j <= 10; j++) {
            cout << setw(2) << decimalToBase18(i * j) << " ";
        }
        cout << endl;
    }
    return 0;
}
